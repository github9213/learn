1. Olvassunk be két egész számot, a program írja ki az összegüket, különbségüket, szorzatukat, hányadosukat.


        #include iostream
        using namespace std;
                            
        int main() {
            int num1, num2;
            cout << "Kerem adja meg az elso szamot: ";
            cin >> num1;
            cout << "Kerem adja meg a masodik szamot: ";
            cin >> num2;
                            
            cout << "A szamok osszege: " << num1 + num2 << endl;
            cout << "A szamok kulonbsege: " << num1 - num2 << endl;
            cout << "A szamok szorzata: " << num1 * num2 << endl;
            cout << "A szamok hanyadosa: " << num1 / num2 << endl;
            return 0;
        }

3. A program olvassa be a téglatest éleit, majd íjra ki a téglatest felszínét és térfogatát.


        #include 
        using namespace std;
                            
        int main() {
            float a, b, c;
            cout << "Kerem adja meg a teglatest egyik oldalat: ";
            cin >> a;
            cout << "Kerem adja meg a teglatest masik oldalat: ";
            cin >> b;
            cout << "Kerem adja meg a teglatest harmadik oldalat: ";
            cin >> c;
                            
            float felszin = 2 * (a * b + b * c + c * a);
            float terfogat = a * b * c;
                            
            cout << "A teglatest felszine: " << felszin << endl;
            cout << "A teglatest terfogata: " << terfogat << endl;
            return 0;
        }

        }

        }

10. Olvasson be egy számot (életkor), majd az értékétől függően írja ki, hogy az illető milyen pozícióban van! (0-6: nem iskolás, 6-14: általános iskolás, 14-18: gimnazista, 18-24: egyetemista, 24-62: dolgozó, egyéb esetben nyugdíjas)


        #include 
        using namespace std;
                            
        int main() {
            int age;
            cout << "Kerem adja meg az eletkorat: ";
            cin >> age;
                            
            if (age >= 0 && age <= 6) {
                cout << "Nem iskolas." << endl;
            } else if (age >= 6 && age <= 14) {
                cout << "Altalanos iskolas." << endl;
            } else if (age >= 14 && age <= 18) {
                cout << "Gimnazista." << endl;
            } else if (age >= 18 && age <= 24) {
                cout << "Egyetemista." << endl;
            } else if (age >= 24 && age <= 62) {
                cout << "Dolgozo." << endl;
            } else {
                cout << "Nyugdijas." << endl;
            }
                            
            return 0;
        }
